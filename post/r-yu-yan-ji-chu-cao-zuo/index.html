R语言基础操作

主要介绍R语言的基本操作以及数据类型和数据结构

1.1 工作空间

getwd() #显示工作空间
setwd() #设置工作空间
ls() #列出当前工作空间中的对象
rm(a, b, ...) #移除一个或多个对象
history(num) #移除一个或多个对象
savehistory("x") #保存命令历史到当前目录下文件下 x 中
loadhistory("x") #载入命令历史
xsave.image("x") #保存工作空间到当前目录下的文件x
q()# 退出

1.2 文件输入输出

打开文件：

file.edit("path", fileEncoding="UTF-8") #用UTF-8编码打开导入 .R 文件，导入R的脚本文件
dev.new() #打开新界面

保存运行结果：

sink("path") #开始行，保存文件到 path 中
x <- 2
y <- x * 2
cat("x=", x, "\n")
cat("y=", y)
sink() #结束行

保存现有对象：

save.image("path") #保存对象
load("path") #加载对象

保存图形文件：

pdf("path") or png("path") or jpeg("path") or bmp("path") #开始行，保存为PDF,png,JPEG,bmp格式

dev.off() #一定要有结束行

运行文件：

source("path", encoding = "UTF-8") #运行path文件，可选择编码格式

常用函数：

list.dirs() #列出path下所有文件夹
list.files() #列出path下所有文件

1.3 数据类型

这些对象中最简单的是向量对象，并且这些原子向量有六种数据类型，也称为六类向量。 其他R对象建立在原子向量之上。

数据类型 例 校验
Logical（逻辑型） T(TRUE), F(FALSE) v <- TRUE print(class(v)) 它产生以下结果[1] "logical"
Numeric（数字） 12.3，5，999 v <- 23.5 print(class(v)) 它产生以下结果 [1] "numeric"
Integer（整型） 2L，34L，0L v <- 2L print(class(v)) 它产生以下结果 [1] “integer”
Character（字符） ‘a’ , ‘“good”, “TRUE”, ’23.4’ v <- "TRUE" print(class(v)) 它产生以下结果 [1] “character”
Complex（复合型） 3 + 2i v <- 2+5i print(class(v)) 它产生以下结果 -[1] "complex"
Raw（原型） "Hello" 被存储为 48 65 6c 6c 6f v <- charToRaw("Hello") print(class(v)) 它产生以下结果 - [1] "raw"

在R编程中，非常基本的数据类型是称为向量的R对象，其保存如上所示的不同类的元素。 请注意，在R中，类的数量不仅限于上述六种类型。 例如，我们可以使用许多原子向量并创建一个数组，其类将成为数组。

# 数据类型判断
is.numeric("")
is.logical("")
...

# 数据类型转换
as.numeric("")
as.character("")
...

1.4 数据结构

1.4.1 Vector 向量

当你想用多个元素创建向量时，你应该使用 c() 函数赋值，这意味着将元素组合成一个向量。

apple <- c('red','green',"yellow") #不同类型会自动向最高类型转换
print(apple)
print(class(apple)) #Get the class of the vector.
# 将得到如下结果：
# [1] "red"  "green"  "yellow"
# [1] "character"

apple <- c(x=2, y=3, z=4) #可直接对向量中的元素命名
names(apple)[2] <- "t" #修改向量元素名

apple <- 1:6 #将1-6赋值给apple
apple + 2 #将apple中每一个元素+2
apple + c(1:4) #将长的向量里每一个元素与短的向量中的每一个元素对应相加，不足从头补齐

seq(from, to, by=) #从from到to以给定步长赋值
seq(from, to, length=) #从from到to生成给的长度向量
rep(data, times) #将data整体重复times遍
rep(data, each=) #将data中每一个元素重复
paste(A, B, sep="") #将A，B字符串

# 作业1：
# 生成如下向量：
# "金融1701甲" "金融1701乙" "金融1702甲" "金融1702乙"
# "金融1703甲" "金融1703乙" "金融1704甲" "金融1704乙"
A <- c("甲", "乙")
result1 <- paste(rep("金融170", 8), rep(1:4, each=2), A, sep = "")
result2 <- paste("金融170", rep(1:4, each=2), A, sep = "")
result1
result2

生成随机向量：

sample(x, size, replase=FALSE, prob=NULL) #产生size个随机整数，数据范围为x
runif(data, min=, max=) #生成data个随机数，min为最小值，max为最大值
rnorm(data, mean=, sd=) #生成服从正态分布的随机数，mean为均值，sd为标准差
rpois(data, lamda) #生成参数为lamda的data个服从泊松分布的随机数
rfunc(n, p1, p2, ...) #func为概率分布函数，n为生成的随机数，p1，p2，...为需要的参数

# 作业2：
# 随机产生一个整数向量，其中负数用1-负数代替，正数用1+正数代替
# 问题，如果先计算负数，则负数结果为正数，后续计算正数时结果受到影响。
# 方法一：先计算正数
a <- sample(-10:10, 5)
a
a[a > 0] <- 1 + a[a > 0]
a[a < 0] <- 1 - a[a < 0]
a
# 方法二：先计算负数，条件使用复制的向量
a1 <- a
a[a1 < 0] <- 1 - a[a1 < 0]
a[a1 > 0] <- 1 + a[a1 > 0]
a

# 随机产生一个整数向量，负数用NA代替，再用0代替NA
b <- sample(-10:10, 6)
b
b[b < 0] <- NA
b[is.na(b)] <- 0
b

1.3.2 Matrices 矩阵

矩阵是二维矩形数据集，可以使用矩阵函数的向量输入创建

matrix(data, nrow, ncol, byrow=, dimnames=list(rownames, colnames))
# data必选，填入矩阵的数据
# nrow可选，表示行数
# ncol可选，表示列数
# byrow可选，默认为F，即按列排列数据，若为T则按行排列数据
# dimnames可选，指定行和列的名称

#指定行和列的名称
rownames(xx) <- value # 行
colnames(xx) <- value # 列

M = matrix(c('a', 'a', 'b', 'c', 'b', 'a'), nrow = 2, ncol = 3, byrow = TRUE)
print(M)

# 结果如下：
# 		[,1] [,2] [,3]
# [1,] "a"  "a"  "b"
# [2,] "c"  "b"  "a"

diag(4) #生成一个四维单位矩阵
diag(1:4) #指定单位矩阵对角线数值

#取值
ma <- matrix(1:16, 4, 4)
ma[1, ] #取第一行数据
ma[ ,1] #取第二行数据
ma[1, 1] #取第一行第一列数据
ma[1, c(2, 3)] #去第一行第二至三列数据
diag(ma) #取对角线数据
diag(diag(ma)) #取对角线数据形成的单位矩阵

#运算
m1 <- matrix(1:12, 3, 4)
m2 <- t(m1)
m3 <- matrix(1:12, 3, 4, byrow = TRUE)

m4 <- m1 %*% m2 #矩阵乘法
m5 <- m1 * m3 #矩阵对应位置数值相乘
m4
m5

crossprod(m1, m3) #m1的转置于m3相乘

m6 <- matrix(c(5:10, 12:21), 4, 4, byrow = T)
solve(m6) #m6的逆矩阵
m6

eigen(m6) #求矩阵特征值和特征向量
lower.tri(m6) #取下三角矩阵，返回逻辑值
upper.tri(m6) #取上三角矩阵，返回逻辑值

#作业
#1. 产生随机数，形成矩阵，再用1-矩阵中的负数，1+矩阵中的正数
ma1 <- matrix(sample(-50:50, 16), 4, 4)
ma1

ma1[ma1 > 0] <- 1 + ma1[ma1 > 0]
ma1[ma1 < 0] <- 1 - ma1[ma1 < 0]
ma1

#2. 取上三角和下三角矩阵
ma1[lower.tri(ma1)] <- 0 #上三角
ma1[upper.tri(ma1)] <- 0 #下三角

ma1[row(ma1) > col(ma1)] <- 0 #上三角
ma1[row(ma1) > col(ma1)] <- 0 #下三角

1.4.3 Lists 列表

列表也是R对象

#list列表里可以是不同的数据类型，也可以是不同的数据结构
x1 <- c(4:18)
x2 <- c("njh", "hehai", "1763710131", "信息管理与信息系统")
x3 <- matrix(1:16, 4, 4)
x4 <- list(x1, x3)
x5 <- list(x1, "信息"=x2, x3, x4)
x5

list1 <- list(c(2, 5, 3), 21.3, sin)
list1

# 结果如下：
# [[1]]
# [1] 2 5 3
# [[2]]
# [1] 21.3
# [[3]]
# function (x)  .Primitive("sin")

#提取元素
x5[[1]][2] #第一个对象的第二个元素
x5$信息 #通过名称提取

x6 <- as.list(x1) #向量转换为列表
x6

x7 <- unlist(x6) #列表转换为向量
x7
